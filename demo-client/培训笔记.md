### 方式一

* 初始化npm

```
npm init
```

* 安装webpack

```
npm install webpack webpack-cli --save-dev
```

* 安装@ccxc/tool

```
npm i @ccxc/tool
```

* 创建index.js

```
import {CommonTool, LoginTool, HttpTool} from "@ccxc/tool";

const appId = 'xc0807916485';
const oauthLogin = 'https://ccxc.vip/BCloud/oauth/login';
const getMyAppToken = "/my_app/token"

LoginTool.initLogin({
    appId: appId,
    loginUrl: oauthLogin,
    getToken(req) {
        return new Promise(function (resolve) {
            const url = CommonTool.analysisParam(getMyAppToken, {code: req['code']});
            HttpTool.get(url).then(res => {
                CommonTool.setCookie('token', res.data.accessToken);
                CommonTool.setCookie('refresh-token', res.data.refreshToken);
                resolve(res.data);
            });
        });
    },
    loginFail(error) {
        return new Promise(function (resolve) {
            if (confirm(error.content)) {
                resolve(true);
            } else {
                resolve(false);
            }
        });
    },
    loginSuccess() {
        CommonTool.jump("/index");
    }
});

```

* 创建webpack.config.js

```
const path = require('path');

module.exports = {
    // 入口文件
    entry: './index.js',
    output: {
        // 输出目录和文件名
        path: path.resolve(__dirname, 'dist'),
        filename: 'bundle.js'
    },
    // 适配 Node 环境（如果是前端浏览器可删除）
    target: 'node',
    mode: 'production' // 生产模式(压缩)，development(不压缩)
};
```

* 配置编译命令

```
"build": "webpack"
```
